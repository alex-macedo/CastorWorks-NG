**State-aware rendering logic:**

```
subscription state → what to show:
  isOnTrial (subscription_tier_id = 'trial', trial_ends_at in future)
    → "Trial" badge + daysRemaining counter + "Upgrade to Paid Plan" CTA
  Trial expired (subscription_tier_id = 'sandbox', came from trial)
    → "Trial Expired" message + "Subscribe Now" CTA
  sandbox (no trial, no payment)
    → "Sandbox" badge + "Start Free Trial" or "Subscribe Now" CTA
  active paid subscription (subscription.status = 'active')
    → Current plan card: tierName, billingPeriod, next renewal date (current_period_end), monthly amount
    → "Change Plan" button → opens SubscriptionCheckoutFlow dialog
    → "Manage Billing" button → calls create-billing-portal-session → window.location.href = url
  cancel_at_period_end = true
    → Show "Cancels on {date}" warning banner within the active plan card
  past_due
    → Show "Payment past due" warning with "Manage Billing" CTA
```

**Subscription page JSX outline:**

```tsx
export default function SubscriptionPage() {
  const { t } = useTranslation('subscription')
  const tenantId = useTenantId()
  const { subscription, isActive, isCancelling, isLoading } = useSubscription()
  const { isOnTrial, daysRemaining } = useTenantTrial()
  const [searchParams, setSearchParams] = useSearchParams()
  const [checkoutOpen, setCheckoutOpen] = useState(false)
  const [portalLoading, setPortalLoading] = useState(false)

  // Handle redirect from Stripe Checkout
  useEffect(() => {
    if (searchParams.get('success') === '1') {
      toast.success(t('paymentSuccess'))
      setSearchParams({})
    } else if (searchParams.get('canceled') === '1') {
      toast.info(t('paymentCanceled'))
      setSearchParams({})
    }
  }, [searchParams, setSearchParams, t])

  const handleManageBilling = async () => {
    setPortalLoading(true)
    try {
      const { data, error } = await supabase.functions.invoke('create-billing-portal-session', {
        body: { tenant_id: tenantId },
      })
      if (error || !data?.url) throw error ?? new Error('No portal URL')
      window.location.href = data.url
    } catch (err) {
      toast.error(String(err))
    } finally {
      setPortalLoading(false)
    }
  }

  if (isLoading) return <p className="text-muted-foreground">{t('loadingSubscription')}</p>

  return (
    <div className="space-y-6">
      <h2 className="text-2xl font-semibold">{t('pageTitle')}</h2>

      {/* Current plan card */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            {t('currentPlan')}
            {isOnTrial && <Badge variant="secondary">{t('trialBadge')}</Badge>}
            {/* show tier name from subscription or useTenantTrial tier */}
          </CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          {/* Render appropriate state-based content per logic above */}
          {/* See detailed rendering logic in action description */}
        </CardContent>
      </Card>

      {/* Checkout dialog */}
      <Dialog open={checkoutOpen} onOpenChange={setCheckoutOpen}>
        <DialogContent className="max-w-3xl">
          <SubscriptionCheckoutFlow
            preselectedTierId={subscription?.tier_id}
            onClose={() => setCheckoutOpen(false)}
          />
        </DialogContent>
      </Dialog>
    </div>
  )
}
```
