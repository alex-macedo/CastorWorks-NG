**Step 1: Create supabase/functions/_shared/ai-metering.ts**

```typescript
import { createServiceRoleClient } from './authorization.ts';

export interface AIActionResult {
  allowed: boolean;
  degraded: boolean;
  remaining: number;
}

export async function consumeAIActions(
  tenantId: string,
  feature: string,
  actionsToConsume: number,
  userId: string,
  modelUsed: string,
  tokensIn = 0,
  tokensOut = 0,
  costBrl = 0,
  cached = false
): Promise<AIActionResult> {
  const supabase = createServiceRoleClient();

  const { data, error } = await supabase.rpc('consume_ai_actions', {
    p_tenant_id: tenantId,
    p_feature: feature,
    p_actions: actionsToConsume,
    p_user_id: userId,
    p_model_used: modelUsed,
    p_tokens_in: tokensIn,
    p_tokens_out: tokensOut,
    p_cost_brl: costBrl,
    p_cached: cached,
  });

  if (error) throw new Error(`AI metering error: ${error.message}`);

  return {
    allowed: data.allowed,
    degraded: data.degraded,
    remaining: data.remaining,
  };
}
```

Note: Cached hits (cached=true) should pass actionsToConsume=0 when called from EFs that use aiCache.ts. The caller handles this logic.

- `metadata.pack_size` = selected pack size (200, 500, or 2000) as string

Full auth flow same as create-checkout-session:
1. authenticateRequest(req) — get user
2. verifyTenantAccess(user.id, tenantId) — confirm user belongs to tenant

5. Return { url: session.url }

case 'checkout.session.completed': {
  const session = event.data.object as Stripe.Checkout.Session;
  // CRITICAL: Only handle action pack purchases; ignore subscription checkouts

        p_tenant_id: tenantId,
        p_credits: packSize,
      });
    }
  }
  break;
}
