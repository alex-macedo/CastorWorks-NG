**File 1: `supabase/functions/create-checkout-session/index.ts`**

Receives POST `{ tenant_id: string, tier_id: string, billing_period: 'monthly' | 'annual' }`.

Logic:
1. Parse and validate request body — return 400 if missing required fields.
2. Call `verifyTenantAccess(req, supabase)` from `_shared/authorization.ts` — return 401/403 if unauthorized.
3. Fetch `stripe_price_id_monthly` or `stripe_price_id_annual` from `subscription_tiers` WHERE `id = tier_id`. If NULL, return 400 `{ error: 'Stripe price not configured for this tier' }`.
4. Look up existing `stripe_customer_id` from `subscriptions` WHERE `tenant_id = tenant_id` (most recent row). If not found, create a new Stripe Customer via `stripe.customers.create({ metadata: { tenant_id } })` and store in a new `subscriptions` row (with `status='incomplete'`).
5. Create Stripe Checkout Session:

```typescript

  customer: stripeCustomerId,
  mode: 'subscription',
  line_items: [{ price: priceId, quantity: 1 }],

  subscription_data: {
    metadata: { tenant_id, tier_id, billing_period },
  },
  success_url: `${Deno.env.get('APP_URL')}/settings/subscription?success=1&session_id={CHECKOUT_SESSION_ID}`,
  cancel_url:  `${Deno.env.get('APP_URL')}/settings/subscription?canceled=1`,
})
```

6. Return `{ url: session.url }`.

Use `import Stripe from 'npm:stripe'` and `import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'`.

**File 1: `src/hooks/useSubscription.ts`**

```typescript
import { useQuery } from '@tanstack/react-query'
import { supabase } from '@/integrations/supabase/client'
import { useTenantId } from '@/contexts/TenantContext' // adjust import to match existing pattern

export interface Subscription {
  id: string
  tenant_id: string
  stripe_customer_id: string | null
  stripe_subscription_id: string | null
  stripe_price_id: string | null
  tier_id: string
  billing_period: 'monthly' | 'annual'
  status: string
  current_period_start: string | null
  current_period_end: string | null
  cancel_at_period_end: boolean
  created_at: string
  updated_at: string
}

export interface UseSubscriptionResult {
  subscription: Subscription | null
  isActive: boolean
  isCancelling: boolean
  currentTier: string | null
  isLoading: boolean
  error: Error | null
}

export function useSubscription(): UseSubscriptionResult {
  const tenantId = useTenantId()

  const { data, isLoading, error } = useQuery({
    queryKey: ['tenant', tenantId, 'subscription'],
    queryFn: async () => {
      if (!tenantId) return null
      const { data, error } = await supabase
        .from('subscriptions')
        .select('*')
        .eq('tenant_id', tenantId)
        .order('created_at', { ascending: false })
        .limit(1)
        .maybeSingle()
      if (error) throw error
      return data as Subscription | null
    },
    enabled: !!tenantId,
  })

  return {
    subscription: data ?? null,
    isActive: data?.status === 'active' || data?.status === 'trialing',
    isCancelling: data?.cancel_at_period_end ?? false,
    currentTier: data?.tier_id ?? null,
    isLoading,
    error: error as Error | null,
  }
}
```
